SELECT * FROM CUSTOMERS EXCEPT SELECT * FROM ITEMS

CREATE VIEW CUSTOMERS_SHOPS_VIEW AS SELECT CUSTOMERS.SID,SHOPS.SID FROM CUSTOMERS.SID=SHOPS.SID
